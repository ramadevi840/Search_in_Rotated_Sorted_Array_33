class Solution {
    public int search(int[] nums, int target) {
        int low=0;
        int right=nums.length-1;
        while(low<=right){
            int mid= low+(right-low)/2;
            if(nums[mid]==target){
                return mid;
            }
// if target is btw low and mid then we check low is lessthen equal to mid  
            else if(nums[low]<=nums[mid]){
// here is elemt is btw low and mid then go left side to mid 
                if(nums[low]<=target && target<nums[mid]){
                    right=mid-1;
                }
                else{
                       low=mid+1;
                }
           
            }
// here is elemt is btw mid and high then go mid to right side 
            else{
                if(nums[mid]<target && target<=nums[right]){
                   low=mid+1;
                }
                else{
                right=mid-1;
                }
            }
        }
        return -1;
    }
}
